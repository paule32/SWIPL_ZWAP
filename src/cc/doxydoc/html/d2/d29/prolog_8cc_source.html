158int
159main(int argc, char** argv)
160{
161 try {
162 if (argc < 3)
163 throw PL_Exception( std::string("too few arguments.") );
164
165 if (std::string(argv[1]).size() < 1)
166 throw PL_Exception( std::string("error: can not open input file.") );
167
168 if (std::string(argv[2]).size() < 1)
169 throw PL_Exception( std::string("error: can not open output file.") );
170
171 #ifdef CHARSET_UTF8
172 iput_file = std::wifstream( std::string( argv[1] ) );
173 oput_file = std::wofstream( argv[2] );
174 #else
175 iput_file = std::ifstream ( std::string( argv[1] ) );
176 oput_file = std::ofstream ( argv[2] );
177 #endif
178
179 iput_file_str = readFile ( std::string( argv[1] ) );
180
181 PL_lookaheadPosition = -1;
182 PL_lineno = 1;
183 PL_nestedComment = 0;
184
185 PL_ident.clear();
186 while (1) {
187 label_start:
188 PL_lookaheadPosition = PL_lookaheadPosition + 1; if (
189 PL_lookaheadPosition >= iput_file_str.size()) break;
190 PL_lookaheadWChar = iput_file_str.at(PL_lookaheadPosition);
191
192 if (PL_lookaheadWChar == 0x00)
193 break;
194
195 if (
196 (PL_lookaheadWChar == ' ' ) ||
197 (PL_lookaheadWChar == '\t')) {
198 continue;
199 }
200
201 if (PL_lookaheadWChar == '\n') {
202 PL_lineno =
203 PL_lineno + 1;
204 continue;
205 }
206
207 if (PL_lookaheadWChar == '/')
208 {
209 PL_lookaheadPosition = PL_lookaheadPosition + 1;
210 PL_lookaheadWChar = iput_file_str.at(
211 PL_lookaheadPosition);
212
213 if (PL_lookaheadPosition >= iput_file_str.size())
214 throw PL_Exception("not yet implemented.");
215
216 if (PL_lookaheadWChar == '*')
217 {
218 PL_nestedComment =
219 PL_nestedComment + 1;
220
221 while (1) {
222 label_comment2:
223 PL_lookaheadPosition = PL_lookaheadPosition + 1; if (
224 PL_lookaheadPosition >= iput_file_str.size())
225 throw PL_Exception("unterminated comment");
226
227 PL_lookaheadWChar = iput_file_str.at(
228 PL_lookaheadPosition);
229
230 if (PL_lookaheadWChar == '\n') {
231 PL_lineno =
232 PL_lineno + 1;
233 continue;
234 }
235
236 if (PL_lookaheadWChar == '/') {
237 PL_lookaheadPosition = PL_lookaheadPosition + 1; if (
238 PL_lookaheadPosition >= iput_file_str.size())
239 throw PL_Exception("unterminated comment");
240
241 PL_lookaheadWChar = iput_file_str.at(
242 PL_lookaheadPosition);
243
244 if (PL_lookaheadWChar == '*') {
245 PL_nestedComment =
246 PL_nestedComment+1;
247 continue;
248 }
249 }
250
251 if (PL_lookaheadWChar == '*') {
252 PL_lookaheadPosition = PL_lookaheadPosition + 1;
253 PL_lookaheadWChar = iput_file_str.at(
254 PL_lookaheadPosition);
255
256 if (PL_lookaheadPosition >= iput_file_str.size())
257 throw PL_Exception("unterminated comment");
258
259 if (PL_lookaheadWChar == '/') {
260 PL_nestedComment =
261 PL_nestedComment-1; if (
262 PL_nestedComment<1)
263 break;
264 }
265 }
266 }
267 } else {
268 // todo
269 throw PL_Exception("not yets implemented.");
270 }
271 }
272
273 // one line comment
274 if (PL_lookaheadWChar == '%') {
275 label_comment:
276 while (1) {
277 PL_lookaheadPosition = PL_lookaheadPosition + 1; if (
278 PL_lookaheadPosition >= iput_file_str.size()) break;
279 PL_lookaheadWChar = iput_file_str.at(
280 PL_lookaheadPosition);
281
282 if (PL_lookaheadWChar == '\n') {
283 PL_lineno =
284 PL_lineno + 1;
285 break;
286 }
287 }
288
289 if (PL_lookaheadPosition >= iput_file_str.size())
290 break;
291 continue;
292 }
293
294 if (
295 (PL_lookaheadWChar >= 'a' && PL_lookaheadWChar <= 'z') ||
296 (PL_lookaheadWChar >= 'A' && PL_lookaheadWChar <= 'A') ||
297 (PL_lookaheadWChar == '_'))
298 {
299 label_ident:
300 PL_ident += PL_lookaheadWChar;
301 while (PL_lookaheadPosition !=
302 iput_file_str.size())
303 {
304 PL_lookaheadPosition = PL_lookaheadPosition + 1;
305 PL_lookaheadWChar = iput_file_str.at(
306 PL_lookaheadPosition);
307
308 if (PL_lookaheadWChar == '\n') {
309 STDCOUT << PL_ident << std::endl;
310 PL_ident.clear();
311 PL_lineno =
312 PL_lineno + 1;
313 break;
314 }
315
316 if (
317 (PL_lookaheadWChar >= 'a' && PL_lookaheadWChar <= 'z') ||
318 (PL_lookaheadWChar >= 'A' && PL_lookaheadWChar <= 'A') ||
319 (PL_lookaheadWChar >= '0' && PL_lookaheadWChar <= '9') ||
320 (PL_lookaheadWChar == '_')) {
321 if (PL_nestedComment < 1) {
322 goto label_ident;
323 }
324 }
325 else {
326 STDCOUT << PL_ident << std::endl;
327 PL_ident.clear();
328 break;
329 }
330 }
331 }
332 }
333 STDCOUT << std::endl;
334 STDCOUT << "Compiled: OK" << std::endl;
335 STDCOUT << "Lines : " << PL_lineno << std::endl;
336 }
337 catch (PL_Exception& e)
338 {
339 STDCOUT << "error : " << PL_lineno
340 << std::endl
341 << "reason: " << e.message
342 << std::endl;
343
344 return PL_fail();
345 } return PL_success();
346}
